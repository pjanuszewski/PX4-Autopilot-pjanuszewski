px4_add_module(
	MODULE examples__lqr_app
	MAIN lqr_app
	SRCS
		lqr_app.cpp
	DEPENDS
		mathlib
		px4_work_queue
		ct_optcon
	)

	${CONTROL_TOOLBOX_PATH}/ct_core/include
	${CONTROL_TOOLBOX_PATH}/ct_optcon/include
)

if(OpenMP_CXX_FOUND)
    target_link_libraries(examples__lqr_app PUBLIC OpenMP::OpenMP_CXX)
endif()
